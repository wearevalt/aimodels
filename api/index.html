Endpoints
GET /v1/models Liste tous les modèles
// Requête curl -X GET https://api.valtlens.com/v1/models \ -H "Authorization: Bearer vlns_votre_cle_api" \ -H "Content-Type: application/json" // Paramètres optionnels ?category=text-to-video // Filtrer par catégorie ?free=true // Modèles gratuits seulement ?limit=20&offset=0 // Pagination // Réponse { "data": [ { "id": "sora", "name": "Sora", "developer": "OpenAI", "rating": 4.9, "categories": ["text-to-video"], "free": false, "url": "https://openai.com/sora" } ], "total": 98, "page": 1 }
GET /v1/models/:id Détail d'un modèle
// Requête curl -X GET https://api.valtlens.com/v1/models/sora \ -H "Authorization: Bearer vlns_votre_cle_api" // Réponse { "id": "sora", "name": "Sora", "developer": "OpenAI", "description": "...", "rating": 4.9, "specs": { "duration": "60s", "resolution": "1080p", "pricing": "ChatGPT Plus" }, "tags": ["Text-to-Video", "High Quality"], "updatedAt": "2025-03-29T00:00:00Z" }
GET /v1/categories Liste des catégories
// Réponse { "categories": [ {"id": "text-to-video", "label": "Text-to-Video", "count": 42}, {"id": "image-to-video", "label": "Image-to-Video", "count": 28}, {"id": "free", "label": "Gratuit / Open Source", "count": 35} ] }
POST /v1/search Recherche avancée
// Requête curl -X POST https://api.valtlens.com/v1/search \ -H "Authorization: Bearer vlns_votre_cle_api" \ -H "Content-Type: application/json" \ -d '{ "query": "avatar lip sync", "filters": { "free": false, "minRating": 4.5 }, "limit": 10 }' // Réponse { "results": [...], "total": 3, "query": "avatar lip sync" }
